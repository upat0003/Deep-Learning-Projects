import numpy as np

def dense_layer(x, num_neurons, name, activation=None):
# Allocating weights for dense layer by giving x as an input, b as bias with number of neurons and w is the init which is... 
# calculated by truncated_normal module
    
    with tf.name_scope(name):
        num_inputs = int(x.get_shape()[1])
        stddev = 2 / np.sqrt(num_inputs)
        init = tf.truncated_normal([num_inputs, num_neurons], stddev=stddev)
        
        W = tf.Variable(init, name="W")
        b = tf.Variable(tf.truncated_normal([num_neurons], stddev=stddev))
        z = tf.matmul(x, W) + b 
    
    #Sigmoid is the activation function which calculates the output from the value given using the formula
    if activation == "sigmoid":        
        return tf.nn.sigmoid(z) 

    #Sigmoid activation function returns the inputted value if possitive or zero if not positive
    elif activation == "relu": 
        return tf.nn.relu(z)
    
    else:
        return z

inputs = tf.placeholder(tf.float32, shape=[None, num_inputs], name="inputs") 
labels = tf.placeholder(tf.int32, shape=[None], name="labels") 


with tf.name_scope("dnn"):
    
    #Going ahead with hidden layers which takes inputs and pass it to the next one after it goes through the activation function
    tf.compat.v1.set_random_seed(29143926)
    hidden1 = dense_layer(inputs, num_hidden1, name= "layer1", activation= "relu") 
    hidden2 = dense_layer(hidden1, num_hidden2, name= "layer2", activation= "relu")
    logits = dense_layer(hidden2, num_outputs, name= "logits")

with tf.name_scope("loss"):
    
    #Counting the entropy and loss values
    xentropy = tf.nn.sparse_softmax_cross_entropy_with_logits(labels=labels, logits=logits, name='xentropy') 

    loss = tf.reduce_mean(xentropy, name="loss")

with tf.name_scope("evaluation"):

    #Calculating the accuracy values using labels as given 
    correct =  tf.nn.in_top_k(logits, labels, 1) 
    accuracy = tf.reduce_mean(tf.cast(correct, tf.float32)) 
    
with tf.name_scope("train"):
    learning_rate = 0.001
    
    #Using Adam optimizer in this case to improve the performance of the model
    tf.compat.v1.set_random_seed(29143926)
    optimizer = tf.train.AdamOptimizer(learning_rate=0.001) 
    grads = optimizer.compute_gradients(loss)
    training_op = optimizer.apply_gradients(grads)
    
    for var in tf.trainable_variables():
        tf.summary.histogram(var.op.name + "/values", var)
        
    for grad, var in grads:
        if grad is not None:
            tf.summary.histogram(var.op.name + "/gradients", grad)

#Summarising the overall accuracy of the model
accuracy_summary = tf.summary.scalar("accuracy", accuracy)
